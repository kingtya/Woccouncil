activity_woc_council = {
	is_shown = {
		has_royal_court = yes
		is_landed = yes
		has_dlc_feature = tours_and_tournaments
		has_dlc_feature = advanced_activities
	}

	can_start_showing_failures_only = {
		#is_available_adult = yes
		custom_description = {
			text = "is_not_convened_recently"
			NAND = { 
				has_character_flag = convened_woc_council_normal_recently
				has_character_flag = convened_woc_council_urgent_recently
			}
		}
		trigger_if = {
			limit = {
				scope:special_option ?= flag:woc_council_normal
			}
			custom_description = {
				text = already_scheming_woc_council_missions
				NOT = {
					any_scheme = {
						scheme_type = woc_council_missions
					}
				}
			}
		}
	}

	is_valid = {
		scope:host = {
			is_alive = yes
			#is_capable_adult = yes
			is_imprisoned = no
			is_landed = yes
			NOT = { is_incapable = yes }
		}
		# If no one shows up
		trigger_if = {
			limit = {
				is_current_phase_active = yes
			}
			has_attending_activity_guests = yes
		}
	}
	
	on_host_death = {
		every_attending_character = {
			limit = { is_alive = yes }
			trigger_event = {
				id = woc_council_activity.3001
				days = 1 # So you don't get spammed
			}
		}
		# Code sends your entourage home on invalidated
		scope:activity.var:host_var = {
			# Saving the title for loc
			primary_title = { save_scope_as = host_title }
			# Inform your heir if you have one
			player_heir ?= {
				trigger_event = {
					id = woc_council_activity.3001
					days = 1
				}
			}
		}
	}

	on_invalidated = {
		#监禁取消了活动
		if = {
			limit = {
				scope:host = { is_imprisoned = yes }
			}
			#Host has been imprisoned
			every_attending_character = {
				limit = {
					NOT = { this = scope:host }
				}
				trigger_event = woc_council_activity.3002
			}
			scope:host = { trigger_event = woc_council_activity.3003 }
		}
		#无能取消了活动
		if = {
			limit = {
				scope:host = { is_incapable = yes }
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
			}
			scope:host = {
				trigger_event = activity_system.0330
			}
			every_attending_character = {
				limit = { NOT = { this = scope:host } }
				trigger_event = activity_system.0331
			}
		}
		#失去头衔取消了活动
		if = {
			limit = {
				scope:host = { is_landed = no }
			}
			scope:activity = {
				activity_type = { save_scope_as = activity_type }
			}
			every_attending_character = {
				trigger_event = activity_system.0320
			}
		}
	}
